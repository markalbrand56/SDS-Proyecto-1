from sklearn.metrics import confusion_matrix

def fp_penalty_metric(y_true, y_pred_proba):
    """
    Mtrica para penalizar los falsos positivos en clasificacin binaria.

    :param y_true: Valores verdaderos de la variable objetivo.
    :param y_pred_proba: Probabilidades predichas de la clase positiva.
    :return: Nombre de la mtrica, valor de la mtrica y si es mayor o menor es mejor.
    """
    # Clasificacin binaria con umbral 0.5
    y_pred = (y_pred_proba >= 0.5).astype(int)
    tn, fp, fn, tp = confusion_matrix(y_true, y_pred).ravel()

    if tp == 0:
        ratio = np.inf
    else:
        ratio = (tp + fp) / tp

    # LightGBM maximiza la mtrica  invertimos
    score = 1 / ratio if ratio != np.inf else 0

    return "fp_penalty_ratio", score, True

def custom_f1(y_true, y_pred_proba):
    """
    Mtrica F1 personalizada para clasificacin binaria.

    :param y_true: Valores verdaderos de la variable objetivo.
    :param y_pred_proba: Probabilidades predichas de la clase positiva.
    :return: Nombre de la mtrica, valor de la mtrica y si es mayor o menor es mejor.
    """
    y_pred = (y_pred_proba >= 0.5).astype(int)
    tn, fp, fn, tp = confusion_matrix(y_true, y_pred).ravel()

    precision = tp / (tp + fp + 1e-6)
    recall = tp / (tp + fn + 1e-6)
    f1 = 2 * (precision * recall) / (precision + recall + 1e-6)

    return "custom_f1", f1, True

def make_recall_high_risk_feval(region_flag, threshold=0.5):
    """
    Funcin para calcular la mtrica de recall en regiones de alto riesgo.

    :param region_flag: Array que indica si la transaccin es de una regin de alto riesgo.
    :param threshold:  Umbral para clasificar una transaccin como fraude.
    :return: Funcin de evaluacin personalizada para LightGBM.
    """
    def recall_high_risk_eval(y_true, y_pred_proba):
        """
        Funcin de evaluacin personalizada para calcular el recall en regiones de alto riesgo.

        :param y_true: Valores verdaderos de la variable objetivo.
        :param y_pred_proba: Probabilidades predichas de la clase positiva.
        :return: Nombre de la mtrica, valor de la mtrica y si es mayor o menor es mejor.
        """
        y_pred = (y_pred_proba >= threshold).astype(int)

        mask = region_flag == 1
        y_true_hr = y_true[mask]
        y_pred_hr = y_pred[mask]

        if y_true_hr.sum() == 0:
            recall_hr = 0.0
        else:
            tp = ((y_pred_hr == 1) & (y_true_hr == 1)).sum()
            fn = ((y_pred_hr == 0) & (y_true_hr == 1)).sum()
            recall_hr = tp / (tp + fn + 1e-6)

        return "recall_high_risk", recall_hr, True

    return recall_high_risk_eval
